import { describe, it, expect } from "vitest";
import { computeTotalCents, applyRefund } from "../src/cents";

describe("checkout money math", () => {
  it("sums line items in integer cents (no float drift)", () => {
    expect(computeTotalCents([{ cents: 199 }, { cents: 850 }, { cents: 1 }])).toBe(1050);
  });
  it("rejects a refund larger than the captured amount", () => {

  });

    expect(applyRefund(1000, 300)).toBe(700);

### src/cents.ts
```
export function computeTotalCents(items: { cents: number }[]): number {
  return items.reduce((sum, i) => sum + i.cents, 0);
}
export function applyRefund(capturedCents: number, refundCents: number): number {
  if (refundCents > capturedCents) throw new Error("refund exceeds captured amount");
  return capturedCents - refundCents;
}
```
